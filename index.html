// 1. Open Box & Play Music

function openBox() {

    const lid = document.getElementById('boxLid');

    const music = document.getElementById('bgMusic');

    

    lid.classList.add('open');

    music.play().catch(e => console.log("Music will play after first interaction"));

}

 

// 2. Popups

function showPopup(text, imgSrc) {

    const popup = document.getElementById('popup');

    document.getElementById('popImg').src = imgSrc;

    document.getElementById('popText').innerText = text;

    popup.style.display = 'flex';

}

function closePopup() {

    document.getElementById('popup').style.display = 'none';

}

 

// 3. Toggle Love Drawer

function toggleDrawer() {

    const drawer = document.getElementById('loveDrawer');

    drawer.classList.toggle('active');

}

 

// 4. Chocolate Rain

function rainEffect() {

    const icons = ['🍫', '💝', '🍭', '💖', '✨'];

    for(let i = 0; i < 40; i++) {

        setTimeout(() => {

            const drop = document.createElement('div');

            drop.style.position = 'fixed';

            drop.style.top = '-50px';

            drop.style.left = Math.random() * 100 + 'vw';

            drop.style.fontSize = (Math.random() * 20 + 20) + 'px';

            drop.innerText = icons[Math.floor(Math.random() * icons.length)];

            drop.style.zIndex = '1000';

            document.body.appendChild(drop);

            

            const anim = drop.animate([

                { transform: 'translateY(0) rotate(0deg)' },

                { transform: `translateY(110vh) rotate(${Math.random() * 360}deg)` }

            ], { duration: 3000, easing: 'linear' });

            

            anim.onfinish = () => drop.remove();

        }, i * 150);

    }

}
